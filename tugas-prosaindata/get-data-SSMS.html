import pyodbc
import pandas as pd

cnxn_str = ("Driver={ODBC Driver 17 for SQL Server};"
            "Server=LAPTOP-SOC00JA3\SQLEXPRESS;" 
            "Database=prosain{210411100112};Trusted_Connection=yes")

cnxn = pyodbc.connect(cnxn_str)
cursor = cnxn.cursor()	
cursor.execute("Select petallength, class From iris")
row = cursor.fetchall() 
cursor.close()

data=[]
for i in row:
    data.append([i[0], i[1].strip('"')])

# Membuat DataFrame dari hasil query
df_sqlServer = pd.DataFrame(data, columns=['petallength', 'class'])

# Menyalin DataFrame ke variabel baru
df3 = df_sqlServer

# Menampilkan data
print(df3)
